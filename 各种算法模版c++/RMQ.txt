	 for(int j=1;j<=n-(1<<i)+1;j++)
      f[j][i]=max(f[j][i-1],f[j+(1<<(i-1))][i-1]);

int query(int x,int y)

	return max(f[x][z],f[y-(1<<z)+1][z]);
